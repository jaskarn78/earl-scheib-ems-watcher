```python
if self.path.split("?")[0] == "/earlscheibconcord/queue/send-now":
    sig = self.headers.get("X-EMS-Signature", "")
    if not _validate_hmac(raw, sig):
        self._send_json(401, {"error": "invalid signature"})
        return
    try:
        body = json.loads(raw.decode("utf-8"))
        job_id = int(body["id"])
    except (ValueError, KeyError, json.JSONDecodeError, UnicodeDecodeError):
        self._send_json(400, {"error": "invalid JSON"})
        return

    con = get_db()
    try:
        cur = con.cursor()
        # Atomic claim: only one concurrent caller can flip this row.
        cur.execute("UPDATE jobs SET sent = 1, sent_at = ? WHERE id = ? AND sent = 0",
                    (int(time.time()), job_id))
        con.commit()
        if cur.rowcount != 1:
            self._send_json(404, {"error": "not_found_or_already_sent"})
            return
        cur.execute("SELECT job_type, phone, name FROM jobs WHERE id = ?", (job_id,))
        row = cur.fetchone()
    finally:
        con.close()

    # Compose SMS body using the existing templates (must match the UI preview).
    if row["job_type"] == "24h":
        sms_body = MSG_24H.format(name=row["name"])
    elif row["job_type"] == "3day":
        sms_body = MSG_3DAY.format(name=row["name"])
    elif row["job_type"] == "review":
        sms_body = MSG_REVIEW.format(name=row["name"])
    else:
        sms_body = ""

    ok = send_sms(row["phone"], sms_body)
    if ok:
        log.info("send-now: id=%s phone=%s type=%s OK", job_id, row["phone"], row["job_type"])
        self._send_json(200, {"sent": True})
    else:
        # Twilio failed: roll back the sent flag so the scheduler can retry later.
        con = get_db()
        try:
            cur = con.cursor()
            cur.execute("UPDATE jobs SET sent = 0, sent_at = 0 WHERE id = ?", (job_id,))
            con.commit()
        finally:
            con.close()
        log.error("send-now: id=%s twilio send failed; rolled back sent flag", job_id)
        self._send_json(500, {"error": "twilio_send_failed"})
    return
```
- Rationale for the rollback: `send_sms` returns bool on transport failure. Without the rollback the row is permanently "sent" without an actual SMS ever leaving Twilio. Matches the spirit of the existing scheduler_loop (which leaves sent=0 on failure for retry).
- NOTE: the "sid" in the response was nice-to-have in the task description, but `send_sms` as written does not return the sid — only bool. Return `{"sent": True}` and revisit sid exposure in a later quick task if needed.
