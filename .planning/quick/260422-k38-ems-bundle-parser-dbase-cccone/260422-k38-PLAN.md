// AD1/VEH/ENV map field-name → string value (trimmed of trailing spaces).

    ENV         map[string]string // E_DOC_NUM / E_RO / E_EST_NUM / E_DOC_ID / E_REF (whichever exists)

}

package scanner

// BundleCandidate is a detected EMS 2.01 dBase bundle ready to settle/parse.
type BundleCandidate struct {
    Basename string            // "G-123" (no extension)
    Files    map[string]string // key: lowercase extension without dot ("ad1","veh","env",...). value: absolute path.
    // Virtual path used for dedup — filepath.Join(dir, basename + ".bundle").
    // NOT a real file on disk. Used only as the `filepath` column value in processed_files.
    VirtualPath string
}

// DetectBundles scans dir for EMS 2.01 dBase bundles.
// A bundle = 2+ files sharing a basename where the set includes BOTH .ad1 AND .veh (case-insensitive).
// Returns candidates in deterministic order (sorted by basename).
func DetectBundles(dir string, logger *slog.Logger) []BundleCandidate {

    // Pre-settle dedup check using the VIRTUAL path + anchor mtime.
    anchor, err := bundleMtimeAnchor(bun.Files)
    if err != nil {
        logger.Error("bundle mtime anchor failed", "basename", bun.Basename, "err", err)
        errors++
        continue
    }
    already, err := db.IsProcessed(cfg.DB, bun.VirtualPath, anchor)
    if err != nil {
        logger.Error("bundle IsProcessed failed", "basename", bun.Basename, "err", err)
    }
    if already {
        logger.Debug("bundle already processed", "basename", bun.Basename)
        continue
    }

    // Settle every component file.
    settled, postAnchor, totalSize := bundleSettle(bun, opts, func(m string, a ...any) { logger.Debug(m, a...) })
    if !settled {
        logger.Info("bundle not settled — will retry next cycle", "basename", bun.Basename)
        continue
    }

    // Re-dedup with post-settle anchor.
    already, _ = db.IsProcessed(cfg.DB, bun.VirtualPath, postAnchor)
    if already {
        logger.Debug("bundle already processed (post-settle)", "basename", bun.Basename)
        continue
    }

    // Parse + synthesize XML.
    bundle, err := ems.ParseBundle(bun.Basename, bun.Files)
    if err != nil {
        logger.Error("ems parse failed", "basename", bun.Basename, "err", err)
        errors++
        continue
    }
    xmlBytes := ems.RenderBMS(bundle)

    // Hash for dedup row.
    sha, err := bundleSHA256(bun.Files)
    if err != nil {
        logger.Error("bundle sha256 failed", "basename", bun.Basename, "err", err)
        errors++
        continue

    // Send. The Sender is injected; the bundle-trigger query-param wrap happens

    if cfg.Sender(bun.VirtualPath, xmlBytes) {

            logger.Error("bundle MarkProcessed failed", "basename", bun.Basename, "err", mErr)
            errors++
            continue
        }
        logger.Info("ems bundle processed", "basename", bun.Basename, "bytes", len(xmlBytes))
        processed++
    } else {
        errors++
    }
