sign.go:
```go
package webhook

import (
    "crypto/hmac"
    "crypto/sha256"
    "encoding/hex"
)

// Sign returns hex HMAC-SHA256 of body using secret as raw UTF-8 key.
// Returns empty string when secret is empty (matches Python `if secret_key:` guard).
// Byte-identical to: hmac.new(secret.encode('utf-8'), body, hashlib.sha256).hexdigest()
func Sign(secret string, body []byte) string {
    if secret == "" {
        return ""
    }
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write(body)
    return hex.EncodeToString(mac.Sum(nil))
}
```

Send uses `net/http.Client{Timeout: cfg.Timeout}`. Manual retry loop:
- `httpAttempts = 3`, `backoffBase = 1.0 * time.Second`
- On each attempt: build request, set headers, call client.Do
- On network error: log warning, if attempt < 3 sleep and double backoff, continue; if attempt == 3 log error, return false
- On 2xx: log info "Sent filename -> url (HTTP N, M bytes)", return true
- On retryable status (408/425/429/5xx): if attempt < 3 log warning with body[:200], sleep, double backoff, continue; if attempt == 3 log error, return false
- On non-retryable: log error with body[:200], return false immediately (do not consume remaining attempts)

```go
// Send POSTs a heartbeat to webhookURL+"/heartbeat" (non-fatal — any error is logged at Debug).
func Send(webhookURL string, secretKey string, logger *slog.Logger) {
    if webhookURL == "" {
        return
    }
    hostname, _ := os.Hostname()
    body := []byte(fmt.Sprintf("<Heartbeat><Host>%s</Host></Heartbeat>", hostname))
    sig := webhook.Sign(secretKey, body)
    req, err := http.NewRequest("POST", webhookURL+"/heartbeat", bytes.NewReader(body))
    if err != nil {
        logger.Debug("heartbeat request build failed", "err", err)
        return
    }
    req.Header.Set("Content-Type", "application/xml")
    req.Header.Set("X-EMS-Signature", sig)
    req.Header.Set("X-EMS-Heartbeat", "1")
    client := &http.Client{Timeout: 10 * time.Second}
    resp, err := client.Do(req)
    if err != nil {
        logger.Debug("heartbeat failed (non-fatal)", "err", err)
        return
    }
    defer resp.Body.Close()
    logger.Debug("heartbeat sent", "status", resp.StatusCode)
}
```
