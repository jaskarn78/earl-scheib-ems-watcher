```go
// DBRetry runs fn, retrying up to DB_LOCK_ATTEMPTS(5) times on "locked" errors.
// Delay starts at 500ms and doubles each attempt (500ms, 1s, 2s, 4s).
func DBRetry(fn func() error, logger *slog.Logger) error {
    delay := 500 * time.Millisecond
    for attempt := 1; attempt <= 5; attempt++ {
        err := fn()
        if err == nil {
            return nil
        }
        if !strings.Contains(strings.ToLower(err.Error()), "locked") || attempt == 5 {
            return err
        }
        logger.Warn("SQLite busy", "attempt", attempt, "of", 5, "retrying_in", delay)
        time.Sleep(delay)
        delay *= 2
    }
    return nil
}
```
